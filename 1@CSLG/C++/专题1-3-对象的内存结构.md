###二 对象引用
//1. &i取地址  2. 3&4 按位与  3 . a>0 && b<0 逻辑与  4 . int &i = j; 定义一个应用类型
	#include <iostream>
	using namespace std;
	
	void change(int &n){
		++n;
	}
	
	class D{
		int data;
	public:
		D(int d = 0):data(d){
			cout<<"C("<<data<<endl;
		};
	
		~D(){
				cout<<"~C("<<data<<endl;
			};
	};
	
	int main(){
		int i = 3;
		int &j = i; //j就是i的引用(别名)，j和i的地址相同,是同一个变量,一个地址映射多个名字
		change(j);
		cout<<"i="<<i<<",j="<<j<<endl;
	
		D c1 = 10; //当类的成员只有一个数据成员时，可以这样写
		//D &c2;   //error
		D &c2 = c1; //引用必须进行初始化,对象的引用不会产生对象的拷贝
	
		return 0;
	}
